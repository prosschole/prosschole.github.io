- katakan = kata dasar 'kata'

- dapatkan = kata dasar 'dapat'

- ambilkan = kata dasar 'ambil'

- bawakan = kata dasar 'bawa'

- bacakan = kata dasar 'baca'

( an ) :

- Mingguan = kata dasar 'minggu'

- Tahunan = kata dasar 'tahun'

- Bulanan = kata dasar 'bulan'

- Harian = kata dasar 'hari'

- ucapan = kata dasar 'ucap'

( i )

- Pukuli = kata dasar 'pukul'

- Saringi = kata dasar 'saring'

- Kantongi = kata dasar 'kantong'
