1) n(s) = 42
n(A) = 30
n(B) = 28
n(A u B)^c = 4
n(A n B) = .......????

jawab :
42 = 30+28 - n(A n B) + 4
42 = 62 - n(A n B)
n(A n B) = 62-42
n(A n B) = 20


2) n(s) = 45

n(B) = 26
n(A u B)^c = 5
n(A n B) = ....... .?????

jawab :
45 = 27+26 - n(A n B) + 5
45 = 58 - n(A n B)
n(A n B) = 58 - 45
n(A n B) = 13
