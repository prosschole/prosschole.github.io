BC = 8 cm

TO = 12 cm

AC^2 = AB^2 + BC^2

AC^2 = 6^2 + 8^2

AC^2 = 36 + 64

AC^2 = 100

AC = V100

AC = 10 cm

TC^2 = OC^2 + TO^2

TC^2 = (1/2 × AC)^2 + TO^2

TC^2 = (1/2 × 10)^2 + 12^2

TC^2 = 5^2 + 12^2

TC^2 = 25 + 144

TC^2 = 169

TC = V169

TC = 13 cm

K = [2 × (AB + BC)] + (4 × TC)

K = [2 × (6 + 8)] + (4 × 13)

K = (2 × 14) + 52

K = 28 + 52

K = 80 cm
